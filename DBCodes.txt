CREATE DATABASE IF NOT EXISTS `consultacnpj`;
USE `consultacnpj` ;

CREATE TABLE IF NOT EXISTS `consultacnpj`.`atributos` (
  `id` INT(11) NOT NULL AUTO_INCREMENT,
  `NomeFantasia` VARCHAR(100) NOT NULL,
  `RazaoSocial` VARCHAR(100) NOT NULL,
  `CNPJ` VARCHAR(20) NOT NULL,
  `CNAE` VARCHAR(200) NOT NULL,
  `CNAECodigo` VARCHAR(7) NOT NULL,
  `CEP` VARCHAR(10) NOT NULL,
  `Abertura` VARCHAR(10) NULL DEFAULT NULL,
  `Telefone` VARCHAR(15) NULL DEFAULT NULL,
  `Email` VARCHAR(50) NOT NULL,
  `Logradouro` VARCHAR(150) NOT NULL,
  `status` VARCHAR(7) NULL DEFAULT NULL,
  PRIMARY KEY (`id`));
